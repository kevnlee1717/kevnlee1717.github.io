# 对异常的封装

  • 流程：在config里让框架的exception handler 设置为我们自己的handler， 自己的handler里主要是 render函数，按我们自己定义的方式来显示异常。

我们自己定义了一个 baseexception 继承自 exception 类，多了 msg(异常消息) code(返回code， 如200，400， 500) error_code(自定义的异常代号)

在我们自己抛出的异常就赋值 baseexception 里的值，然后我们自己的 exception handler 在 render的时候就可以按照这些自定义的字段来返回显示给客户端。


// 自定义的 exception 比系统exception多了 三个字段
class BaseException extends Exception{

    public $msg = "异常"; //默认消息为异常
    public $code = 400; //默认错误代码为 400
    public $error_code = 10000; //默认自定义异常代码为10000
    
    public function __construct($params = []){
        if(!is_array($params)) return; //如果不是数组初始化， 格式不符合，直接使用默认值

        //如果三个字段里哪个存在 就给哪个赋值，不然就用默认值
        if(array_key_exists('msg', $params)) $this->msg = $params['msg'];
        if(array_key_exists('code', $params)) $this->code = $params['code'];
        if(array_key_exists('error_code', $params)) $this->error_code = $params['error_code'];
    }

}

//自定义的 exception handler

class ExceptionHandler extends Handle{
    public $msg;
    public $code;
    public $error_code;

    function render(\Exception $e)
    {
        //如果是自定义的 exception 
        if($e instanceof BaseException){
            $this->code = $e->code;
            $this->error_code = $e->error_code;
            $this->msg = $e->msg;
        }else{
            // 如果是系统的异常，而且系统打开了debug
            if(config('app_debug')) return parent::render($e);

            //没有打开debug的情况下，全部显示未知错误
            $this->code = 500;
            $this->error_code = 10000;
            $this->msg = "未知错误";
        }

        $ret = [
            'msg' => $this->msg,
            'error_code' => $this->error_code,
        ];

        // 错误代码直接在第二个参数里返回。会显示在http的head里。

        return json($ret, $this->code);

    }

}
