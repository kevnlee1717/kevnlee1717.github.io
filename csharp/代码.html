# 读取一个进程中的内存数据

  • process类的 getprocess()获取所有的process 根据进程名字获得到相应进程。

  • 获得到进程后可以列出所有加载的模块，找到感兴趣的模块，可以读取版本号等，各种信息。

  • 用readprocessmemory函数去读取， 这个函数是vc函数，c#要调用要做一个特定的声明和转换。

  • 从指针里读取字符串的时候，先从地址里读出字符串的地址，然后再读取，，地址读取的时候，注意高位的要写在最前面。

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Diagnostics;
using System.Drawing;
using System.Linq;
using System.Runtime.InteropServices;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace WindowsFormsApp1
{
    public partial class Form1 : Form
    {
        Process wxProcess = null;
        ProcessModule wxModule = null;

        IntPtr baseAddress = (IntPtr)0;

        public Form1()
        {
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            this.textBox1.Clear();

            Process[] processes = Process.GetProcesses();

            foreach (Process process in processes)
            {
                if (process.ProcessName == "WeChat")
                {
                    this.wxProcess = process;
                    this.textBox1.AppendText("微信找到" + Environment.NewLine);
                    foreach (ProcessModule module in process.Modules)
                    {
                        if (module.ModuleName == "WeChatWin.dll")
                        {
                            this.wxModule = module;
                            this.textBox1.AppendText("WeChatWin.dll找到" + Environment.NewLine);
                            this.baseAddress = this.wxModule.BaseAddress;
                            this.textBox1.AppendText("WeChatWin.dll 基址: 0x" + ((int)this.baseAddress).ToString("x2") + Environment.NewLine);
                            this.textBox1.AppendText("WeChatWin.dll 版本: " + this.wxModule.FileVersionInfo.FileVersion.ToString() + Environment.NewLine);
                            

                            this.textBox1.AppendText("微信名: " + this.readProcessString(wxProcess.Handle, this.baseAddress + 0x162094C) + Environment.NewLine);
                            //1620C14
                            this.textBox1.AppendText("微信头像: " + this.readProcessString(wxProcess.Handle, (IntPtr)this.getAddress(wxProcess.Handle, (IntPtr)(this.baseAddress + 0x1620C14))) + Environment.NewLine);

                            break;
                        }

                    }
                    if (this.wxModule == null)
                    {
                        this.textBox1.AppendText("WeChatWin.dll没找到" + Environment.NewLine);

                    }
                    break;
                }
            }

            if (this.wxProcess == null)
            {
                this.textBox1.AppendText("微信没找到！" + Environment.NewLine);
            }
        }


        private String readProcessString(IntPtr hProcess, IntPtr lpBaseAddress, int iReadSize = 100)
        {
            byte[] bfRead = new byte[iReadSize];

            if(ReadProcessMemory(hProcess, lpBaseAddress, bfRead, iReadSize, 0) == false)
            {
                this.textBox1.AppendText("读取进程数据时候出错！" + Environment.NewLine);
            }

            String result = "";
            String tempString = Encoding.UTF8.GetString(bfRead);
            foreach(char temp in tempString)
            {
                if(temp == '\0')
                {
                    break;
                }
                result += temp.ToString();
            }

            return result;
        }

        public int getAddress(IntPtr hProcess, IntPtr lpBaseAddress)
        {

            byte[] bfRead = new byte[4];

            if (ReadProcessMemory(hProcess, lpBaseAddress, bfRead, 4, 0) == false)
            {
                this.textBox1.AppendText("读取进程数据时候出错！" + Environment.NewLine);
            }

            String sAddress = bfRead[3].ToString("x2") +
                 bfRead[2].ToString("x2") +
                 bfRead[1].ToString("x2") +
                 bfRead[0].ToString("x2");

            return int.Parse(sAddress, System.Globalization.NumberStyles.HexNumber);

        }


        [DllImport("kernel32.dll", SetLastError = true)]
        static extern bool ReadProcessMemory(
        IntPtr hProcess,
        IntPtr lpBaseAddress,
        [Out] byte[] lpBuffer,
        int dwSize,
        int lpNumberOfBytesRead);





    }
}
